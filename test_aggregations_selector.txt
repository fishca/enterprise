// 1) Numeric array with projection
var nums = New Array
nums.Add(10)
nums.Add(20)
nums.Add(30)

Message("=== T1 selector x => x * 2 ===")
Message(nums.Sum(Function(x) Return x * 2 EndFunction))      // 120
Message(nums.Min(Function(x) Return x * 2 EndFunction))      // 20
Message(nums.Max(Function(x) Return x * 2 EndFunction))      // 60
Message(nums.Average(Function(x) Return x * 2 EndFunction))  // 40

Message(orders.Max(Function(o) Return o.Amount EndFunction))      // 250
Message(orders.Average(Function(o) Return o.Amount EndFunction))  // 133.33...
