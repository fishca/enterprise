// Typed-enum dispatch smoke — verify TYPE_ITERATOR migration
// didn't break OPER_FOREACH / OPER_NEXT_ITER paths.
//
// Plain iteration через arr — under the hood OPER_FOREACH calls
// AsIterator() instead of ConvertToType<ibValueIterator>(). Single
// int compare on m_typeClass + static_cast, без RTTI walk через
// wxClassInfo.

var arr = New Array
arr.Add(10)
arr.Add(20)
arr.Add(30)

var sum = 0
For Each x In arr Do
    sum = sum + x
EndDo
Message(sum)   // expect 60

// LINQ pipeline tail also uses iterator dispatch:
var result = arr.Where(Function(x) Return x >= 20 EndFunction).ToArray()
Message("filtered count=" + result.Count())  // expect 2
For Each v In result Do
    Message(v)
EndDo
// expect 20, 30
