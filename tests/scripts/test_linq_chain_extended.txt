// Chain-syntax LINQ — OrderBy / OrderByDescending / GroupBy / Join.
// Lazy state classes (ibValueOrderByState / GroupByState / JoinState)
// materialise on first MoveNext, emit ibValueQuery wrapping pipeline.

Message("=== T1 OrderBy ===")
var nums = New Array
nums.Add(3)
nums.Add(1)
nums.Add(4)
nums.Add(1)
nums.Add(5)
nums.Add(9)
nums.Add(2)
nums.Add(6)

var sorted = nums.OrderBy(Function(x) Return x EndFunction).ToArray()
For Each v In sorted Do
    Message(v)   // 1 1 2 3 4 5 6 9
EndDo

Message("=== T2 OrderByDescending ===")
var desc = nums.OrderByDescending(Function(x) Return x EndFunction).ToArray()
For Each v In desc Do
    Message(v)   // 9 6 5 4 3 2 1 1
EndDo

Message("=== T3 GroupBy ===")
var orders = New Array
orders.Add(New Structure("Country, Amount", "USA",   100))
orders.Add(New Structure("Country, Amount", "UK",     50))
orders.Add(New Structure("Country, Amount", "USA",   200))
orders.Add(New Structure("Country, Amount", "FR",     75))
orders.Add(New Structure("Country, Amount", "UK",     25))

var groups = orders.GroupBy(Function(o) Return o.Country EndFunction).ToArray()
Message("group count=" + groups.Count())
For Each g In groups Do
    Message(g.Key + " items=" + g.Values.Count())
EndDo

Message("=== T4 Chain: OrderBy + Where + Select ===")
var pipeline = nums.OrderBy(Function(x) Return x EndFunction).Where(Function(x) Return x > 2 EndFunction).Select(Function(x) Return x * 10 EndFunction).ToArray()
For Each v In pipeline Do
    Message(v)   // 30 40 50 60 90
EndDo

Message("=== T5 Join ===")
var customers = New Array
customers.Add(New Structure("Id, Name", 1, "Alice"))
customers.Add(New Structure("Id, Name", 2, "Bob"))
customers.Add(New Structure("Id, Name", 3, "Carol"))

var sales = New Array
sales.Add(New Structure("CustId, Amount", 1, 100))
sales.Add(New Structure("CustId, Amount", 2, 200))
sales.Add(New Structure("CustId, Amount", 1, 300))

var joined = customers.Join(sales, Function(c) Return c.Id EndFunction, Function(s) Return s.CustId EndFunction, Function(c, s) Return c.Name + ":" + s.Amount EndFunction).ToArray()
Message("join count=" + joined.Count())
For Each r In joined Do
    Message(r)   // Alice:100, Bob:200, Alice:300
EndDo

Message("=== T6 Closure capture in selector ===")
var threshold = 3
var sortedFiltered = nums.OrderBy(Function(x) Return x EndFunction).Where(Function(x) Return x >= threshold EndFunction).ToArray()
For Each v In sortedFiltered Do
    Message(v)   // 3 4 5 6 9
EndDo

Message("=== suite done ===")
