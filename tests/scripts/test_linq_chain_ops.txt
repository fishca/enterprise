// Chain LINQ extended ops (Skip/Take/Reverse/set ops/terminal/indexed)

var nums = New Array
nums.Add(1)
nums.Add(2)
nums.Add(3)
nums.Add(4)
nums.Add(5)
nums.Add(6)
nums.Add(7)
nums.Add(8)
nums.Add(9)
nums.Add(10)

Message("=== Skip/Take ===")
For Each v In nums.Skip(3).ToArray() Do
    Message(v)   // 4 5 6 7 8 9 10
EndDo
Message("---")
For Each v In nums.Take(3).ToArray() Do
    Message(v)   // 1 2 3
EndDo
Message("---")
For Each v In nums.Skip(2).Take(3).ToArray() Do
    Message(v)   // 3 4 5
EndDo

Message("=== SkipWhile/TakeWhile ===")
For Each v In nums.SkipWhile(Function(x) Return x < 5 EndFunction).ToArray() Do
    Message(v)   // 5 6 7 8 9 10
EndDo
Message("---")
For Each v In nums.TakeWhile(Function(x) Return x < 5 EndFunction).ToArray() Do
    Message(v)   // 1 2 3 4
EndDo

Message("=== Reverse ===")
For Each v In nums.Take(5).Reverse().ToArray() Do
    Message(v)   // 5 4 3 2 1
EndDo

Message("=== Concat/Union/Intersect/Except ===")
var a = New Array; a.Add(1); a.Add(2); a.Add(3); a.Add(4)
var b = New Array; b.Add(3); b.Add(4); b.Add(5); b.Add(6)
For Each v In a.Concat(b).ToArray() Do
    Message(v)   // 1 2 3 4 3 4 5 6
EndDo
Message("---")
For Each v In a.Union(b).ToArray() Do
    Message(v)   // 1 2 3 4 5 6
EndDo
Message("---")
For Each v In a.Intersect(b).ToArray() Do
    Message(v)   // 3 4
EndDo
Message("---")
For Each v In a.Except(b).ToArray() Do
    Message(v)   // 1 2
EndDo

Message("=== Last/LastOrDefault/Single/SingleOrDefault/FirstOrDefault ===")
Message(nums.Last())                                            // 10
Message(nums.Take(0).LastOrDefault())                           // empty/default
Message(nums.Take(1).Single())                                  // 1
Message(nums.Take(0).SingleOrDefault())                         // empty
Message(nums.Take(0).FirstOrDefault())                          // empty

Message("=== ElementAt ===")
Message(nums.ElementAt(0))                                      // 1
Message(nums.ElementAt(5))                                      // 6
Message(nums.ElementAtOrDefault(100))                           // empty (out of range)

Message("=== Contains/SequenceEqual ===")
Message(nums.Contains(5))                                       // true
Message(nums.Contains(100))                                     // false

var c = New Array; c.Add(1); c.Add(2); c.Add(3); c.Add(4)
Message(a.SequenceEqual(c))                                     // true (a = [1,2,3,4], c = [1,2,3,4])
Message(a.SequenceEqual(b))                                     // false

Message("=== Aggregate ===")
Message(nums.Aggregate(0, Function(acc, x) Return acc + x EndFunction))  // 55 (sum)
Message(nums.Aggregate(1, Function(acc, x) Return acc * x EndFunction))  // 3628800 (factorial)

Message("=== WhereIndexed/SelectIndexed ===")
For Each v In nums.WhereIndexed(Function(x, i) Return i Mod 2 = 0 EndFunction).ToArray() Do
    Message(v)   // 1 3 5 7 9 (even-indexed elements)
EndDo
Message("---")
For Each v In nums.Take(5).SelectIndexed(Function(x, i) Return "" + i + ":" + x EndFunction).ToArray() Do
    Message(v)   // 0:1 1:2 2:3 3:4 4:5
EndDo
