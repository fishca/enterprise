// ======================================================================
// T01 — simplest: from + select
// Expected: 1 2 3 4 5
// ======================================================================
Message("=== T01: from + select ===");
var nums = New Array;
nums.Add(1); nums.Add(2); nums.Add(3); nums.Add(4); nums.Add(5);
var q01 = from x in nums select x;
foreach (v in q01) { Message(v); }

// ======================================================================
// T02 — where filter
// Expected: 3 4 5
// ======================================================================
Message("=== T02: where ===");
var q02 = from x in nums where x >= 3 select x;
foreach (v in q02) { Message(v); }

// ======================================================================
// T03 — let intermediate
// Expected: 1 4 9 16 25
// ======================================================================
Message("=== T03: let ===");
var q03 = from x in nums

// ======================================================================
// T10 — select { ... } anonymous structure projection
// Expected: lines like "1 -> 1", "2 -> 4", ...
// Note: leading "" forces string-mode concat — ibValue's `+` for
// NUMBER + STRING tries to parse STRING as NUMBER (number-mode) and
// fails on " -> ".
// ======================================================================
Message("=== T10: select { } ===");
var q10 = from x in nums select { N = x, Sq = x * x };
foreach (row in q10) { Message("" + row.N + " -> " + row.Sq); }

// Expected:

arr1.Add(New Structure("Id, Name", 1, "A"));

// ======================================================================
// T14 — JOIN + nested FROM (validates per-level state refactor)
// Expected: 4 rows
//   Alice: Apple / Alice: Banana / Bob: Cherry / Alice: Date
// ======================================================================
Message("=== T14: join + nested from ===");
var apples = New Array; apples.Add("Apple"); apples.Add("Banana");
var cherries = New Array; cherries.Add("Cherry");
var dates = New Array; dates.Add("Date");
var orders = New Array;
orders.Add(New Structure("CustId, Items", 1, apples));
orders.Add(New Structure("CustId, Items", 2, cherries));
orders.Add(New Structure("CustId, Items", 1, dates));
var customers = New Array;
customers.Add(New Structure("Id, Name", 1, "Alice"));
customers.Add(New Structure("Id, Name", 2, "Bob"));
var q14 = from o in orders
          join c in customers on o.CustId equals c.Id
          from item in o.Items

foreach (s in q14) { Message(s); }

// ======================================================================
// T15 — GROUP BY simple (terminal)
// Expected: 3 groups (FR/UK/USA), Container sorted by key
//   FR: 1 / UK: 2 / USA: 3
// ======================================================================
Message("=== T15: group by ===");
var people = New Array;
people.Add(New Structure("Country, Name", "USA", "Alice"));
people.Add(New Structure("Country, Name", "UK",  "Carol"));
people.Add(New Structure("Country, Name", "USA", "Bob"));
people.Add(New Structure("Country, Name", "FR",  "Frank"));
people.Add(New Structure("Country, Name", "UK",  "Dan"));
people.Add(New Structure("Country, Name", "USA", "Eve"));
var q15 = from p in people group p.Name by p.Country;
Message("group count: " + q15.Count());
foreach (g in q15) {
    Message(g.Key + ": " + g.Values.Count());
}

// ======================================================================
// T16 — GROUP BY + Sum via g.Values.Sum() (aggregation per group)
// Expected:
//   FR amount: 100
//   UK amount: 50
//   USA amount: 75
// ======================================================================
Message("=== T16: group + sum ===");
var transactions = New Array;
transactions.Add(New Structure("Country, Amount", "USA", 25));
transactions.Add(New Structure("Country, Amount", "UK",  30));
transactions.Add(New Structure("Country, Amount", "USA", 50));
transactions.Add(New Structure("Country, Amount", "FR",  100));

    Message(g.Key + " amount: " + g.Values.Sum());

        Message("  " + n);
